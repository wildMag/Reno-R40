Code javascript spécifique pour associer et générer le datepicker 
<script language="JavaScript" type="text/javascript">
$('#linkedDates').datepicker({
  beforeShow: readLinked, 
  onSelect: updateLinked
});
  
// Prepare to show a date picker linked to three select controls
function readLinked() {
  $('#linkedDates').val(
    $('#selectDay').val() + '-' +
    $('#selectMonth').val() + '-' + 
    $('#selectYear').val()
  );
  return {};
}
  
// Update three select controls to match a date picker selection
function updateLinked(date) {
  $('#selectDay').val(date.substring(0, 2));
  $('#selectMonth').val(date.substring(3, 5));
  $('#selectYear').val(date.substring(6, 10));
}
  
$('#selectMonth, #selectYear').change(checkLinkedDays);


// Prevent selection of invalid dates through the select controls
function checkLinkedDays() {
  var daysInMonth = 32 - new Date($('#selectYear').val(), $('#selectMonth').val() - 1, 32).getDate();
  $('#selectDay option').attr('disabled', '');
  $('#selectDay option:gt(' + (daysInMonth - 1) +')').attr('disabled', 'disabled');
  if ($('#selectDay').val() > daysInMonth) {
    $('#selectDay').val(daysInMonth);
  }
}
</script>

Code javascript spécifique pour associer et générer les datepickers 
<script language="JavaScript" type="text/javascript">
   $('.dateRange').datepicker({beforeShow: customRange, firstDay: 1, changeFirstDay: false, showStatus: true }); 
   // Customize two date pickers to work as a date range 
   function customRange(input) { 
      return {minDate: (input.id == 'agenda_au' ? $('#agenda_du').datepicker('getDate') : null), 
         maxDate: (input.id == 'agenda_du' ? $('#agenda_au').datepicker('getDate') : null)}; 
   } 
</script>
